## useClaims

Hook for fetching the available claims for an account.

### Import

```ts
import { useClaims } from "@metrom-xyz/react";
```

### Usage

```tsx
import { useClaims } from "@metrom-xyz/react";

function App() {
  const { loading, claims } = useClaims({
    address: "0x0000000000000000000000000000000000000000",
  });

  if (loading) return <div>loading...</div>;

  if (claims.length === 0) return <div>no claims</div>;

  return (
    <div>
      {claims.map((claims) => (
        <div key={claims.id}>
          <p>{claims.campaignId}</p>
          <p>{claims.token.symbol}</p>
          <p>{claims.amount.formatted}</p>
        </div>
      ))}
    </div>
  );
}
```
